def perceptron(X,y):
    w1=w2=b=1 #initial values w1 w2 b 
    learning_rate=0.1
    
    for j in range(1000):
         
        for i in range(X.shape[0]):
            #loss function formula
            z = w1*X[i][0] + w2*X[i][1] + b # fxi main
            
            if z*y[i] < 0:
                w1 =w1+learning_rate*y[i]*X[i][0]
                w2 = w2+learning_rate*y[i]*X[i][1]
                b = b+learning_rate*y[i]
                
    return w1,w2,b
